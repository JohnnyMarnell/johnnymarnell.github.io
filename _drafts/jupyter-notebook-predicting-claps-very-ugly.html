def poor_mans_pivot_table(df, key, more_aggs, ignore):
    cols = [ col for col in list(df) if col != key and col not in more_aggs.keys() and col not in ignore.split(',') ]
    aggs = { col: lambda x: list(x) for col in cols }
    aggs.update(more_aggs)
    collapsed = []
    for idx, row in df.groupby(key).agg(aggs).iterrows():
        g_row = { key: idx }
        for col in cols:
            g_row.update({f'{col}_{i}': val for i, val in enumerate(row[col])})
        g_row.update({agg_key: row[agg_key] for agg_key in more_aggs.keys()})
        collapsed.append(g_row)
    return pd.DataFrame(collapsed)
